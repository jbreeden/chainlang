take

returns at most "count" elements
expect(

    from([1, 2, 3]).take(2)

).to.eql([1, 2]);
returns all the elements if count is higher than _subject.length
expect(

    from([1, 2, 3]).take(5)

).to.eql([1, 2, 3]);

take.all

returns all of the elements
expect(

    from([1, 2, 3]).take.all()

).to.eql([1, 2, 3]);

take.first

return the first element
expect(

    from([1, 2, 3]).take.first()

).to.eql(1);

take.last

return the last element
expect(

    from([1, 2, 3]).take.last()

).to.eql(3);

where

filters the elements of _subject by it's "cond" argument
expect(

    from([1, 2, 3])
    .where(function(el){return el % 2 == 0;})
    .take.all()

).to.eql([2]);

select

accepts a projector function to map each element to a new object
expect(

    from([1, 2, 3])
    .select(function(el){
        return el * 2;
    }).take.all()

).to.eql([2, 4, 6]);
accepts any number of string arguments to project by field names
expect(
    from([
        { name: 'john', id: 1},
        { name: 'larry', id: 2}
    ])
    .select("name")
    .take.all()
).to.eql([{name: 'john'}, {name: 'larry'}]);

union

accepts any number of array arguments and concatenates all of their elements onto _subject
expect(

    from([1, 2])
    .union([3, 4], [5, 6])
    .take.all()

).to.eql([1, 2, 3, 4, 5, 6]);

left.join(right).on(key)

makes a left outter join of _subject with "right" where _subject[key] and right[key] match
var result = 
    from(left)
    .left.join(right)
    .on('id')
    .take.all();

expect(result.length).to.be(2);
result.forEach(function(el){
    if(el.left.id == 1){
        expect(el.right.id).to.be(1);
    } else if (el.left.id == 2){
        expect(el.right).to.eql(null);
    }
    else{
        expect().fail("Join result should only have elements with left.id 1|2");
    }
});
